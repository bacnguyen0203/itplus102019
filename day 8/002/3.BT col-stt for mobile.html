        mobile ( col-* )

        case 1 : tạo giao diện 2 cột => 12/(số cột) = 6 => class css : col-6
        case 2 : tạo giao diện 1 cột => 12/(số cột) = 12 => class css : col-12
        case 3 : tạo giao diện 3 cột => 12/(số cột) = 4 => class css : col-4
        case 4 : tạo giao diện 4 cột => 12/(số cột) = 3 => class css : col-3
        case 5 : tạo giao diện 6 cột => 12/(số cột) = 2 => class css : col-2
        case 6 : tạo giao diện 12 cột => 12/(số cột) = 1 => class css : col-1
